Python 示例 
import requests

url = "https://api.pangolinfo.com/serp/ai-overview"
params = {
    "api_key": "YOUR_API_KEY",
    "query": "best amazon scraping tools",
    "location": "United States",
    "device": "desktop",
    "language": "en"
}

response = requests.get(url, params=params)
data = response.json()

# 获取 AI Overview 内容
ai_overview = data['ai_overview']['content']
sources = data['ai_overview']['sources']

print(f"AI Overview: {ai_overview}")
print(f"Referenced Sources: {sources}")
